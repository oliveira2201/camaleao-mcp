    <script>
        let allConversations = [];
        let currentContact = null;

        // Login
        function login() {
            const email = document.getElementById('email').value;
            const password = document.getElementById('password').value;

            if (email === 'we@we.com' && password === '123456') {
                localStorage.setItem('auth', 'ok');
                document.getElementById('login').classList.add('hidden');
                document.getElementById('app').classList.remove('hidden');
                loadConversations();
            } else {
                document.getElementById('error').style.display = 'block';
            }
        }

        function logout() {
            localStorage.removeItem('auth');
            location.reload();
        }

        // Check auth
        if (localStorage.getItem('auth') === 'ok') {
            document.getElementById('login').classList.add('hidden');
            document.getElementById('app').classList.remove('hidden');
            loadConversations();
        }

        // Carregar conversas
        async function loadConversations() {
            try {
                const response = await fetch('https://n8n.gestorconecta.com.br/webhook/historico-whatsapp?action=conversations');
                const data = await response.json();

                if (data.success) {
                    // Remove registros sem remote_jid (linhas sujas do banco)
                    allConversations = (data.conversations || []).filter(
                        conv => conv.remote_jid && conv.remote_jid.trim() !== ''
                    );
                    displayContacts(allConversations);
                }
            } catch (error) {
                console.error('Erro:', error);
                document.getElementById('contacts-list').innerHTML =
                    '<div class="loading"><p>Erro ao carregar conversas</p></div>';
            }
        }

        // Exibir contatos
        function displayContacts(conversations) {
            const list = document.getElementById('contacts-list');

            if (conversations.length === 0) {
                list.innerHTML = '<div class="loading"><p>Nenhuma conversa encontrada</p></div>';
                return;
            }

            const sanitized = conversations.filter(c => c.remote_jid && c.remote_jid.trim() !== '');

            if (sanitized.length === 0) {
                list.innerHTML = '<div class="loading"><p>Nenhuma conversa encontrada</p></div>';
                return;
            }

            list.innerHTML = sanitized.map(conv => `
                <div class="contact-item" onclick="openChat('${conv.remote_jid}', this)">
                    <div class="avatar">${getInitials(conv.sender_nome)}</div>
                    <div class="contact-info">
                        <div class="contact-name">${conv.sender_nome || 'Sem nome'}</div>
                        <div class="contact-preview">${escapeHtml(conv.preview || '').substring(0, 50)}</div>
                    </div>
                    <div class="contact-meta">
                        <div>${formatDate(conv.ultima_msg_em)}</div>
                        <span class="message-count">${conv.total_mensagens}</span>
                    </div>
                </div>
            `).join('');
        }

        // Filtrar contatos
        function filterContacts() {
            const search = document.getElementById('search').value.toLowerCase();
            const filtered = allConversations.filter(conv =>
                conv.sender_nome?.toLowerCase().includes(search) ||
                conv.remote_jid?.toLowerCase().includes(search)
            );
            displayContacts(filtered);
        }

        // Abrir chat
        async function openChat(remoteJid, el = null) {
            currentContact = remoteJid;

            // Marcar como ativo
            document.querySelectorAll('.contact-item').forEach(item => item.classList.remove('active'));
            if (el) el.classList.add('active');

            // Esconder empty state e mostrar chat
            document.getElementById('empty-chat').classList.add('hidden');
            document.getElementById('chat-view').classList.remove('hidden');

            // Atualizar header
            const conv = allConversations.find(c => c.remote_jid === remoteJid);
            if (conv) {
                document.getElementById('chat-avatar').textContent = getInitials(conv.sender_nome);
                document.getElementById('chat-name').textContent = conv.sender_nome || 'Sem nome';
                document.getElementById('chat-phone').textContent = formatPhone(remoteJid);
            }

            // Carregar mensagens
            try {
                const response = await fetch(`https://n8n.gestorconecta.com.br/webhook/historico-whatsapp?action=messages&remote_jid=${encodeURIComponent(remoteJid)}`);
                const data = await response.json();

                if (data.success) {
                    console.log('Total de mensagens:', data.messages.length);
                    console.log('Primeiras 5 mensagens:', data.messages.slice(0, 5).map(m => ({
                        conteudo: m.conteudo?.substring(0, 20),
                        is_from_me: m.is_from_me,
                        remote_jid: m.remote_jid
                    })));
                    displayMessages(data.messages);
                }
            } catch (error) {
                console.error('Erro:', error);
                document.getElementById('chat-messages').innerHTML =
                    '<div class="loading"><p>Erro ao carregar mensagens</p></div>';
            }
        }

        // Exibir mensagens
        function displayMessages(messages) {
            const container = document.getElementById('chat-messages');

            if (messages.length === 0) {
                container.innerHTML = '<div class="loading"><p>Nenhuma mensagem</p></div>';
                return;
            }

            console.log('sender_nome √∫nicos:', [...new Set(messages.map(m => m.sender_nome))]);

            container.innerHTML = messages.map(msg => {
                // Determina se √© mensagem enviada (suas) ou recebida (do cliente)
                // Normaliza o nome removendo acentos para compara√ß√£o
                const nomeSemAcentos = msg.sender_nome ?
                    msg.sender_nome.normalize('NFD').replace(/[\u0300-\u036f]/g, '').toLowerCase() : '';

                const isFromMe = msg.is_from_me === true || msg.is_from_me === 't' || msg.is_from_me === 'true' ||
                                 nomeSemAcentos.includes('camalao') ||
                                 nomeSemAcentos.includes('camaleao') ||
                                 nomeSemAcentos.includes('prefeitura') ||
                                 nomeSemAcentos.includes('wellington');

                return `
                <div class="message ${isFromMe ? 'sent' : 'received'}">
                    <div class="message-bubble">
                        <div class="message-content">
                            ${msg.tipo_mensagem !== 'text' ? `<span class="media-icon">${getTypeIcon(msg.tipo_mensagem)}</span>` : ''}
                            ${escapeHtml(msg.conteudo || '')}
                        </div>
                        <div class="message-time">${formatTime(msg.enviado_em)}</div>
                    </div>
                </div>
                `;
            }).join('');

            // Scroll para o final
            container.scrollTop = container.scrollHeight;
        }

        // Helpers
        function getInitials(name) {
            if (!name) return '?';
            const parts = name.trim().split(' ');
            if (parts.length === 1) return parts[0][0].toUpperCase();
            return (parts[0][0] + parts[parts.length - 1][0]).toUpperCase();
        }

        function formatDate(dateStr) {
            const date = new Date(dateStr);
            const today = new Date();
            const yesterday = new Date(today);
            yesterday.setDate(yesterday.getDate() - 1);

            if (date.toDateString() === today.toDateString()) {
                return formatTime(dateStr);
            } else if (date.toDateString() === yesterday.toDateString()) {
                return 'Ontem';
            } else {
                return date.toLocaleDateString('pt-BR', { day: '2-digit', month: '2-digit' });
            }
        }

        function formatTime(dateStr) {
            return new Date(dateStr).toLocaleTimeString('pt-BR', {
                hour: '2-digit',
                minute: '2-digit'
            });
        }

        function formatPhone(remoteJid) {
            const phone = remoteJid.replace('@s.whatsapp.net', '');
            if (phone.length === 13) {
                return `+${phone.substring(0, 2)} ${phone.substring(2, 4)} ${phone.substring(4, 9)}-${phone.substring(9)}`;
            }
            return phone;
        }

        function getTypeIcon(type) {
            const icons = { text: 'üí¨', audio: 'üé§', image: 'üì∑', video: 'üé•', document: 'üìÑ', sticker: 'üòä' };
            return icons[type] || 'üìù';
        }

        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }

        // Enter to login
        document.addEventListener('keypress', (e) => {
            if (e.key === 'Enter' && !document.getElementById('login').classList.contains('hidden')) {
                login();
            }
        });
    </script>
